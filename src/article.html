/* Вешаю мою функцию, на стандартную, через её хук */
add_filter('post_gallery', 'gallery_slider', 10, 2);
function gallery_slider($output, $attr) {
	return gallery_template($output, $attr, 'slider');
}

/* Пишу функцию, перехватывающую стандартный вывод галереи */
function gallery_template($output, $attr, $slug) {
	$post = get_post();
	if (!preg_match("/{$slug}/i", $post->post_name )) return $output;
	
	$ids = explode(',', $attr['ids']);
	$images = get_posts(array(
		'include' => $ids,
		'post_status' => 'inherit',
		'post_type' => 'attachment',
		'post_mime_type' => 'image',
		'orderby' => 'post__in',
	));
	if ($images) {
		ob_start();
		include "template-parts/gallery-{$slug}.php";
		$output = ob_get_clean();
		return $output;
	}
}
